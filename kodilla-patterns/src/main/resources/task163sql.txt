
CREATE TABLE ISSUESLISTS (
  ID INT PRIMARY KEY,
  NAME VARCHAR(255)
);

CREATE TABLE ISSUES (
  ID INT PRIMARY KEY,
  ISSUESLIST_ID INT,
  SUMMARY VARCHAR(255),
  DESCRIPTION VARCHAR(255),
  USER_ID_ASSIGNEDTO INT,
  FOREIGN KEY (ISSUESLIST_ID) REFERENCES ISSUESLISTS(ID),
  FOREIGN KEY (USER_ID_ASSIGNEDTO) REFERENCES USERS(ID)
);
